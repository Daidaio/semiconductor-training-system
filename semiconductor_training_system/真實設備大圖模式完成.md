def _get_cooling_status(self, state):
    flow = state.get('cooling_flow', 5.0)
    temp = state.get('lens_temp', 23.0)

    if flow < 4.5 or temp > 25:
        return {"color": "#ef4444", "severity": "critical"}  # 紅色
    elif flow < 4.8 or temp > 24:
        return {"color": "#f59e0b", "severity": "warning"}   # 橙色
