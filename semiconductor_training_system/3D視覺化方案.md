```python
# -*- coding: utf-8 -*-
"""
3D設備視覺化器 (Three.js 版本)
使用 Three.js 渲染真實 DUV 曝光機 3D 模型
"""

from typing import Dict, List
import json

class ThreejsEquipmentVisualizer:
    """Three.js 3D 設備視覺化器"""

    def __init__(self):
        """初始化 3D 視覺化器"""
        self.component_positions = {
            'lens_system': {'x': 0, 'y': 8, 'z': 0},
            'light_source': {'x': 0, 'y': 12, 'z': 0},
            'wafer_stage': {'x': 0, 'y': -2, 'z': 0},
            'reticle_stage': {'x': 0, 'y': 5, 'z': 0},
            'cooling_system': {'x': 6, 'y': 4, 'z': -3},
            'vacuum_chamber': {'x': 0, 'y': 0, 'z': 0},
            'alignment_system': {'x': -5, 'y': 6, 'z': 2},

        生成 3D 設備視圖

            faults: 故障零件列表

        Returns:
            Three.js HTML 視覺化
        """

        # 判斷各零件狀態
        component_status = self._analyze_component_status(state, faults)

        # 生成 JSON 配置
        config_json = json.dumps({
            'components': component_status,
            'camera_position': {'x': 20, 'y': 15, 'z': 20}
        })

        html = f"""

    </style>
</head>
<body>
    <div id="canvas-container"></div>

    <!-- 資訊面板 -->
    <div id="info-panel">
        <div style="font-size: 16px; font-weight: bold; margin-bottom: 10px; color: #64B5F6;">
            🔬 DUV 曝光機 3D 監控
        </div>
        <div id="status-list"></div>
    </div>

    <!-- 控制說明 -->
    <div id="controls">
        <b>🎮 控制說明</b><br>
        • 滑鼠左鍵拖曳：旋轉視角<br>
        • 滑鼠滾輪：縮放<br>
        • 點擊零件：查看詳細資訊

    <!-- Three.js Library -->
    <script src="https://cdnjs.cloudflare.com/ajax/libs/three.js/r128/three.min.js"></script>
    <script src="https://cdn.jsdelivr.net/npm/three@0.128.0/examples/js/controls/OrbitControls.js"></script>

    <script>
        // 配置數據
        const config = {config_json};

        // 初始化場景
        let scene, camera, renderer, controls;
        let componentMeshes = {{}};

        function init() {{
            // 創建場景
            scene = new THREE.Scene();
            scene.background = new THREE.Color(0x0a0e1a);
            scene.fog = new THREE.Fog(0x0a0e1a, 30, 100);

            // 創建相機
            camera = new THREE.PerspectiveCamera(
                60,
                window.innerWidth / 700,
                0.1,
                1000
            );
            camera.position.set(
                config.camera_position.x,
                config.camera_position.y,
                config.camera_position.z
            );

            // 創建渲染器
            const container = document.getElementById('canvas-container');
            renderer = new THREE.WebGLRenderer({{ antialias: true }});
            renderer.setSize(container.clientWidth, container.clientHeight);
            renderer.shadowMap.enabled = true;
            container.appendChild(renderer.domElement);

            // 添加軌道控制器
            controls = new THREE.OrbitControls(camera, renderer.domElement);
            controls.enableDamping = true;
            controls.dampingFactor = 0.05;

            // 添加光源
            addLights();

            // 創建設備模型
            createEquipmentModel();

            // 創建地板網格
            createFloorGrid();

            // 更新狀態面板
            updateStatusPanel();

            // 開始動畫循環
            animate();
        }}

        function addLights() {{
            // 環境光
            const ambientLight = new THREE.AmbientLight(0x404040, 1.5);
            scene.add(ambientLight);

            // 主光源
            const mainLight = new THREE.DirectionalLight(0xffffff, 0.8);
            mainLight.position.set(10, 20, 10);
            mainLight.castShadow = true;
            scene.add(mainLight);

            // 補光
            const fillLight = new THREE.DirectionalLight(0x6699ff, 0.4);
            fillLight.position.set(-10, 10, -10);
            scene.add(fillLight);
        }}

        function createEquipmentModel() {{
            // === 真空腔體 (基座) ===
            const chamberGeometry = new THREE.CylinderGeometry(5, 5, 8, 32);
            const chamberMaterial = new THREE.MeshPhongMaterial({{
                color: 0x2c3e50,
                metalness: 0.8,
                roughness: 0.3,
                transparent: true,
                opacity: 0.7
            }});
            const chamber = new THREE.Mesh(chamberGeometry, chamberMaterial);
            chamber.position.set(0, 0, 0);
            chamber.userData = {{ name: 'vacuum_chamber', label: '真空腔體' }};
            scene.add(chamber);
            componentMeshes['vacuum_chamber'] = chamber;

            // === Wafer 載台 ===
            const stageGeometry = new THREE.CylinderGeometry(2.5, 2.5, 0.5, 32);
            const stageMaterial = new THREE.MeshPhongMaterial({{
                color: 0x34495e,
                metalness: 0.9
            }});
            const stage = new THREE.Mesh(stageGeometry, stageMaterial);
            stage.position.set(0, -2, 0);
            stage.userData = {{ name: 'wafer_stage', label: 'Wafer 載台' }};
            scene.add(stage);
            componentMeshes['wafer_stage'] = stage;

            // === Wafer (晶圓) ===
            const waferGeometry = new THREE.CylinderGeometry(2, 2, 0.05, 32);
            const waferMaterial = new THREE.MeshPhongMaterial({{
                color: 0x95a5a6,
                metalness: 1.0,
                emissive: 0x2980b9,
                emissiveIntensity: 0.2
            }});
            const wafer = new THREE.Mesh(waferGeometry, waferMaterial);
            wafer.position.set(0, -1.7, 0);
            scene.add(wafer);

            // === Reticle 載台 ===
            const reticleStageGeometry = new THREE.BoxGeometry(3, 0.3, 3);
            const reticleStageMaterial = new THREE.MeshPhongMaterial({{
                color: 0x7f8c8d,
                metalness: 0.7
            }});
            const reticleStage = new THREE.Mesh(reticleStageGeometry, reticleStageMaterial);
            reticleStage.position.set(0, 5, 0);
            reticleStage.userData = {{ name: 'reticle_stage', label: 'Reticle 載台' }};
            scene.add(reticleStage);
            componentMeshes['reticle_stage'] = reticleStage;

            // === 投影鏡頭組 ===
            const lensGroup = new THREE.Group();
            lensGroup.position.set(0, 8, 0);

            // 鏡頭外殼
            const lensHousingGeometry = new THREE.CylinderGeometry(1.5, 1.5, 4, 32);
            const lensHousingMaterial = new THREE.MeshPhongMaterial({{
                color: 0x1a1a2e,
                metalness: 0.9,
                roughness: 0.2
            }});
            const lensHousing = new THREE.Mesh(lensHousingGeometry, lensHousingMaterial);
            lensGroup.add(lensHousing);

            // 鏡頭玻璃
            const lensGlassGeometry = new THREE.CylinderGeometry(1.2, 1.2, 0.5, 32);
            const lensGlassMaterial = new THREE.MeshPhongMaterial({{
                color: 0x3498db,
                transparent: true,
                opacity: 0.5,
                metalness: 0.1,
                emissive: 0x3498db,
                emissiveIntensity: 0.3
            }});
            const lensGlass = new THREE.Mesh(lensGlassGeometry, lensGlassMaterial);
            lensGlass.position.y = -1.5;
            lensGroup.add(lensGlass);

            lensGroup.userData = {{ name: 'lens_system', label: '投影鏡頭' }};
            scene.add(lensGroup);
            componentMeshes['lens_system'] = lensGroup;

            // === 光源系統 ===
            const lightSourceGeometry = new THREE.BoxGeometry(3, 2, 3);
            const lightSourceMaterial = new THREE.MeshPhongMaterial({{
                color: 0x16a085,
                metalness: 0.6,
                emissive: 0x27ae60,
                emissiveIntensity: 0.2
            }});
            const lightSource = new THREE.Mesh(lightSourceGeometry, lightSourceMaterial);
            lightSource.position.set(0, 12, 0);
            lightSource.userData = {{ name: 'light_source', label: '光源系統' }};
            scene.add(lightSource);
            componentMeshes['light_source'] = lightSource;

            // === 冷卻系統 (側邊管路) ===
            const coolingGroup = new THREE.Group();

            const coolingBoxGeometry = new THREE.BoxGeometry(2, 3, 2);
            const coolingBoxMaterial = new THREE.MeshPhongMaterial({{

            const coolingBox = new THREE.Mesh(coolingBoxGeometry, coolingBoxMaterial);
            coolingGroup.add(coolingBox);

            // 管路
            const pipeGeometry = new THREE.CylinderGeometry(0.2, 0.2, 8, 16);
            const pipeMaterial = new THREE.MeshPhongMaterial({{

            const pipe1 = new THREE.Mesh(pipeGeometry, pipeMaterial);

            componentMeshes['cooling_system'] = coolingGroup;

            // === 對準系統 (雷射感測器) ===

            componentMeshes['alignment_system'] = alignment;

            // 為每個零件添加點擊事件
            Object.values(componentMeshes).forEach(mesh => {{
                mesh.callback = () => onComponentClick(mesh.userData.name);
            }});

            // 根據狀態添加告警標示
            updateComponentStatus();
        }}

        function updateComponentStatus() {{
            Object.entries(config.components).forEach(([name, status]) => {{
                const mesh = componentMeshes[name];
                if (!mesh) return;

                // 移除舊的告警光暈
                mesh.children = mesh.children.filter(child => !child.userData.isAlarm);

                // 如果是故障，添加紅色閃爍光暈
                if (status.status === 'critical' || status.status === 'warning') {{
                    const alarmColor = status.status === 'critical' ? 0xff0000 : 0xff9800;

                    // 創建光暈
                    const glowGeometry = new THREE.SphereGeometry(2, 32, 32);
                    const glowMaterial = new THREE.MeshBasicMaterial({{
                        color: alarmColor,
                        transparent: true,
                        opacity: 0.3,
                        side: THREE.BackSide
                    }});
                    const glow = new THREE.Mesh(glowGeometry, glowMaterial);
                    glow.userData.isAlarm = true;
                    mesh.add(glow);

                    // 添加閃爍動畫
                    mesh.userData.alarmAnimation = true;
                }}
            }});
        }}

        function createFloorGrid() {{
            const gridHelper = new THREE.GridHelper(30, 30, 0x2c3e50, 0x1a1a2e);
            gridHelper.position.y = -6;
            scene.add(gridHelper);
        }}

        function updateStatusPanel() {{
            const statusList = document.getElementById('status-list');
            let html = '';

            Object.entries(config.components).forEach(([name, info]) => {{
                const statusClass = info.status || 'normal';
                const emoji = statusClass === 'critical' ? '🔴' :
                             statusClass === 'warning' ? '🟡' : '🟢';

                html += `
                    <div class="status-item ${{statusClass}}">
                        ${{emoji}} ${{info.label}}<br>
                        <small style="color: #aaa;">${{info.value}}</small>
                    </div>
                `;
            }});

            statusList.innerHTML = html;
        }}

        function onComponentClick(componentName) {{
            console.log('Clicked:', componentName);
            const info = config.components[componentName];
            if (info) {{
                alert(`零件: ${{info.label}}\\n狀態: ${{info.status}}\\n數值: ${{info.value}}`);
            }}
        }}

        function animate() {{
            requestAnimationFrame(animate);

            // 更新控制器
            controls.update();

            // 閃爍動畫
            Object.values(componentMeshes).forEach(mesh => {{
                if (mesh.userData.alarmAnimation) {{
                    const alarmGlow = mesh.children.find(child => child.userData.isAlarm);
                    if (alarmGlow) {{
                        alarmGlow.material.opacity = 0.2 + Math.sin(Date.now() * 0.005) * 0.15;
                    }}
                }}
            }});

            // 渲染
            renderer.render(scene, camera);
        }}

        // 響應式調整
        window.addEventListener('resize', () => {{
            const container = document.getElementById('canvas-container');
            camera.aspect = container.clientWidth / container.clientHeight;
            camera.updateProjectionMatrix();
            renderer.setSize(container.clientWidth, container.clientHeight);
        }});

        // 啟動
        init();
    </script>
</body>
</html>
        """

        return html

    def _analyze_component_status(self, state: Dict, faults: List[str] = None) -> Dict:
        """分析各零件狀態"""

        components = {}

        # 冷卻系統
        cooling_flow = state.get('cooling_flow', 5.0)
        cooling_status = self._get_status(abs(cooling_flow - 5.0) / 5.0, 0.05, 0.15)
        components['cooling_system'] = {{
            'label': '冷卻系統',
            'value': f'{{cooling_flow:.2f}} L/min',
            'status': cooling_status
        }}

        # 鏡頭溫度
        lens_temp = state.get('lens_temp', 23.0)
        temp_status = self._get_status(abs(lens_temp - 23.0), 0.2, 0.5)
        components['lens_system'] = {{
            'label': '投影鏡頭',
            'value': f'{{lens_temp:.1f}} °C',
            'status': temp_status
        }}

        # 真空壓力
        vacuum = state.get('vacuum_pressure', 1.0e-6)
        vacuum_status = self._get_status((vacuum - 1.0e-6) / 1.0e-6, 0.5, 1.0)
        components['vacuum_chamber'] = {{
            'label': '真空腔體',
            'value': f'{{vacuum:.2e}} Torr',
            'status': vacuum_status
        }}

        # 光源強度
        light = state.get('light_intensity', 100.0)
        light_status = self._get_status(abs(light - 100.0) / 100.0, 0.05, 0.10)
        components['light_source'] = {{
            'label': '光源系統',
            'value': f'{{light:.1f}} %',
            'status': light_status
        }}

        # Wafer 載台
        alignment_error = state.get('alignment_error', 0.0)
        align_status = self._get_status(alignment_error, 2.0, 5.0)
        components['wafer_stage'] = {{
            'label': 'Wafer 載台',
            'value': f'{{alignment_error:.1f}} nm',
            'status': align_status
        }}

        # Reticle 載台
        components['reticle_stage'] = {{
            'label': 'Reticle 載台',
            'value': '正常運作',
            'status': 'normal'
        }}

        # 對準系統
        components['alignment_system'] = {{
            'label': '對準系統',
            'value': f'誤差 {{alignment_error:.1f}} nm',
            'status': align_status
        }}

        return components

    def _get_status(self, deviation: float, warning_threshold: float, critical_threshold: float) -> str:
        """根據偏差量判斷狀態"""
        if deviation >= critical_threshold:
            return 'critical'
        elif deviation >= warning_threshold:
            return 'warning'
        else:
            return 'normal'
```

## 💻 實作方案 2：使用真實 3D 模型（Blender）

如果你想要更真實的視覺效果，可以：

### 第一步：取得或建立 DUV 曝光機 3D 模型

**選項 A：使用現成模型**
1. 從 **Sketchfab** 或 **TurboSquid** 購買/下載半導體設備模型
2. 搜尋關鍵字：`"lithography machine"`, `"semiconductor equipment"`, `"ASML scanner"`

**選項 B：自己用 Blender 建模**（推薦，更精確）
1. 安裝 Blender（免費）
2. 參考 ASML 官方照片建模
3. 導出為 `.gltf` 或 `.glb` 格式（Three.js 支援）

### 第二步：載入真實模型到 Three.js

```javascript
// 使用 GLTFLoader 載入模型
```

```python
            'color': 'red' if state['lens_temp'] > 23.5 else 'blue'
        }},
        {{
            'name': 'Wafer載台',
            'x': 0, 'y': 0, 'z': -2,
            'size': 3,
            'color': 'red' if state['alignment_error'] > 5 else 'green'
        }},
        # ... 其他零件
    ]

    fig = go.Figure()

    for comp in components:
        fig.add_trace(go.Scatter3d(
            x=[comp['x']],
            y=[comp['y']],
            z=[comp['z']],
            mode='markers+text',
            marker=dict(size=comp['size']*10, color=comp['color']),
            text=comp['name'],
            name=comp['name']
        ))

    fig.update_layout(
        scene=dict(
            xaxis=dict(range=[-10, 10]),
            yaxis=dict(range=[-10, 10]),
            zaxis=dict(range=[-5, 15]),
        ),
        title='DUV 曝光機 3D 監控'
    )

    return fig
```
